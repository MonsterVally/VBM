ITA_missiolini_progress_in_ethiopia_success_effect = {
	add_stability = 0.03
	add_war_support = 0.03
	army_experience = 15
	add_command_power = 15
	#Effects in Ethiopia
	if = {
		limit = {
			has_dlc = "By Blood Alone"
		}
		effect_tooltip = {
			#With BBA this is handled in Ethiopian decisions
			ETH = {
				add_war_support = -0.1
			}
		}
	}
	if = {
		#First mission to succeed
		limit = {
			NOT = {
				has_country_flag = ITA_missiolini_progress_in_ethiopia_one_mission_successful_flag
			}
		}
		set_country_flag = ITA_missiolini_progress_in_ethiopia_one_mission_successful_flag
	}
	else_if = {
		#Second mission to succeed
		limit = {
			NOT = {
				has_completed_focus = ITA_the_abyssinian_fiasco
			}
			NOT = {
				has_completed_focus = ITA_struggle_in_ethiopia
			}
		}
		if = {
			limit = {
				has_game_rule = {
					rule = vbm_true_non_hist
					option = VBM_TRUE_NON_HIST__NO
				}
			}
			complete_national_focus = ITA_solid_progress
		}
		else_if = {
			limit = {
				OR = {
					has_country_flag = vbm_ita_true_non_hist_com
					has_country_flag = vbm_ita_true_non_hist_com_dem
				}
			}
			complete_national_focus = ITA_the_abyssinian_fiasco
		}
		else_if = {
			limit = {
				OR = {
					has_country_flag = vbm_ita_true_non_hist_fas
				}
			}
			complete_national_focus = ITA_solid_progress
		}
		else_if = {
			limit = {
				OR = {
					has_country_flag = vbm_ita_true_non_hist_alt_fas
					has_country_flag = vbm_ita_true_non_hist_mon_dem
					has_country_flag = vbm_ita_true_non_hist_mon_non
				}
			}
			complete_national_focus = ITA_struggle_in_ethiopia
		}
		else = {
			complete_national_focus = ITA_solid_progress
		}
	}
}
